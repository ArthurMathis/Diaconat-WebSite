CREATE TABLE Recruteurs (
    Id_recruteurs INTEGER AUTO_INCREMENT PRIMARY KEY, 
    Nom_Recruteurs VARCHAR(64) NOT NULL, 
    Prenom_Recruteurs VARCHAR(64) NOT NULL
);

CREATE TABLE Avoir_rendez_vous_avec (
    Cle_Candidats INTEGER NOT NULL,
    Cle_Recruteurs INTEGER NOT NULL,
    Cle_Instants INTEGER NOT NULL,

    FOREIGN KEY (Cle_Candidats) REFERENCES Candidats(Id_Candidats),
    FOREIGN KEY (Cle_Recruteurs) REFERENCES Recruteurs(Id_Recruteurs),
    FOREIGN KEY (Cle_Instants) REFERENCES Instants(Id_Instants)
);
