DELIMITER $$

CREATE PROCEDURE Insert_Postuler_a()
BEGIN
    DECLARE i INT DEFAULT 0;
    DECLARE candidat_id INT;
    DECLARE instant_id INT;
    
    WHILE i < 50 DO
        -- Sélection aléatoire d'un candidat et d'un instant existant
        SELECT Id_Candidats INTO candidat_id FROM Candidats ORDER BY RAND() LIMIT 1;
        SELECT Id_Instants INTO instant_id FROM Instants ORDER BY RAND() LIMIT 1;
        
        -- Insertion dans la table Postuler_a
        -- Utilisation de INSERT IGNORE pour éviter les doublons
        INSERT IGNORE INTO Postuler_a (Cle_Candidats, Cle_Instants) VALUES (candidat_id, instant_id);
        
        SET i = i + 1;
    END WHILE;
END $$

DELIMITER ;

-- Appel de la procédure pour insérer les données
CALL Insert_Postuler_a();


-- Génération des candidatures pour chaque item de la table Postuler_a
INSERT INTO Candidatures (Status_Candidatures, Cle_Candidats, Cle_Instants, Cle_Sources, Cle_Postes)
SELECT 
    'non-traitee' AS Status_Candidatures,
    pa.Cle_Candidats,
    pa.Cle_Instants,
    FLOOR(RAND() * (SELECT COUNT(*) FROM sources)) + 1 AS Cle_Sources,
    FLOOR(RAND() * (SELECT COUNT(*) FROM postes)) + 1 AS Cle_Postes
FROM 
    Postuler_a pa;
